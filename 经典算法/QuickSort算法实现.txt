void quickSort(int s[] , int l, int r){
    if(s == NULL)
        return;
    int i,j,x;
    if(l < r){
        i = l;
        j = r;
        x = s[i];
        while(i < j){
            while(i < j && s[j] > x)
                j--;
            if(i < j)
                s[i++] = s[j];
            while(i < j && s[i] < x)
                i++;
            if(i < j)
                s[j--] = s[i];
        }
        s[i] = x;
        quickSort(s , l , i - 1);
        quickSort(s , i + 1 , r);
    }
}
void quickSort_2(int s[] , int low , int high){
    int i;
    int last;  //记录基准的位置
    if(low < high){  //当数组中的元素个数大于1时，才进行操作
        last = low; //选取第一个元素作为基准
        //把小于基准元与大于基准元的分开，last记录它们分开的界限
        for(i = low + 1 ; i <= high ;i++){
            if(s[i] < s[low]){
                swap(s[++last] , s[i]);
            }
        }
        swap(s[last] , s[low]);//基准元与界限交换，这样的话，基准元两边就是一边大于，一边小于；
        quickSort_2(s , low , last - 1);
        quickSort_2(s , last + 1 , high);
    }
}
